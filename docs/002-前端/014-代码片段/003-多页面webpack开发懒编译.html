const execa = require("execa")
const express = require("express")
const server = express()
const proxy = require("http-proxy-middleware")
const path = require("path")
const inquirer = require("inquirer")

let port = 6500
const pages = require('../src/page').map(page => {
  return {
    name: page.pageName,
    title: page.title,
    port: port++,
    isStart: false
  }
})

function compile(page) {
  page.isStart = true
  execa(
    `webpack-dev-server --inline  --config build/webpack.dev.conf.js --env.pageName ${page.name} --env.port ${page.port}`, [], {
      stdio: 'inherit'
    }
  )
}

const promptList = [{
  type: "checkbox",
  message: "选择要编译的页面(未选择的页面将在访问时编译): ",
  name: "pages",
  choices: pages
}];

inquirer.prompt(promptList).then(selectPages => {
  const pageMap = {}
  pages.forEach(page => {
    pageMap[page.name] = page
  })
  selectPages.pages.forEach(pageName => {
    const page = pageMap[pageName]
    if (page) {
      compile(page)
    }
  })

  server.listen(8080, err => {
    if (err) {
      console.log(err)
    } else {
      console.log("当前采用懒编译模式，只有在访问时才会进行编译。")
    }
  })
})
